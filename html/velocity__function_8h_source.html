176  virtual bool is_motion_finished(DeformableSimplicialComplex& dsc)
177  {
178  if(time_step < MAX_TIME_STEPS)
179  {
180  for (auto nit = dsc.nodes_begin(); nit != dsc.nodes_end(); nit++)
181  {
182  if(dsc.is_movable(nit.key()))
183  {
184  bool match = false;
185  for (int i = 0; i+2 < pos_old.size(); i += 3)
186  {
187  if (Util::distance_point_triangle<real>(nit->get_pos(), pos_old[i], pos_old[i+1], pos_old[i+2]) < ACCURACY)
188  {
189  match = true;
190  break;
191  }
192  }
193  if (!match) {
194  std::cout << "Stopping criteria: Position " << nit->get_pos() << " has moved." << std::endl;
195  pos_old = dsc.get_interface_face_positions();
196  return false;
197  }
198  }
199  }
200  pos_old = dsc.get_interface_face_positions();
201  }
202  return true;
203  }
